28  //arrays x[1..n] and w[1..n] of length n, containing the abscissas and weights of the Gauss-
29  //Legendre n-point quadrature formula.
30  void gauleg(double x1, double x2, t_dArray1D *x, t_dArray1D *w, int n)
31  {
32  int m,j,i;
33  double z1,z,xm,xl,pp,p3,p2,p1;
34  //High precision is a good idea for this routine.
35  m=(n+1)/2;
36  //The roots are symmetric in the interval, so we only have to Ô¨Ånd half of them.
37  xm=0.5*(x2+x1);
38  xl=0.5*(x2-x1);
39  for (i=1;i<=m;i++) {
40  //Loop over the desired roots.
41  z=cos(3.141592654*(i-0.25)/(n+0.5));
42  //Starting with the above approximation to the ith root, we enter the main loop of reÔ¨Ånement by Newton‚Äôs method.
43  do {
44  p1=1.0;
45  p2=0.0;
46  for (j=1;j<=n;j++) {
47  //Loop up the recurrence relation to get the
48  p3=p2;
49  //Legendre polynomial evaluated at z.
50  p2=p1;
51  p1=((2.0*j-1.0)*z*p2-(j-1.0)*p3)/j;
52  }
53  //p1 is now the desired Legendre polynomial. We next compute pp, its derivative,
54  //by a standard relation involving also p2, the polynomial of one lower order.
55  pp=n*(z*p1-p2)/(z*z-1.0);
56  z1=z;
57  z=z1-p1/pp;
58 
59  //Newton‚Äôs method.
60  } while (fabs(z-z1) > 3.0e-11);
61  (*x)[i]=xm-xl*z;
62  //std::cout << (*x)[i] << std::endl;
63  //Scale the root to the desired interval,
64  (*x)[n+1-i]=xm+xl*z;
65  //and put in its symmetric counterpart.
66  (*w)[i]=2.0*xl/((1.0-z*z)*pp*pp);
67  //Compute the weight
68  (*w)[n+1-i]=(*w)[i];
69  //and its symmetric counterpart.
70  }
71  }
