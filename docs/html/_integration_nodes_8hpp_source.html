38  //Given the lower and upper limits of integration x1 and x2, and given n, this routine returns
39  //arrays x[1..n] and w[1..n] of length n, containing the abscissas and weights of the Gauss-
40  //Legendre n-point quadrature formula.
41  void gauleg(double x1, double x2, t_dArray1D &x, t_dArray1D &w, int n) {
42  int m, j, i;
43  double z1, z, xm, xl, pp, p3, p2, p1;
44  m = (n + 1) / 2;
45  xm = 0.5 * (x2 + x1);
46  xl = 0.5 * (x2 - x1);
47  for (i = 1; i <= m; i++) {
48  z = cos(3.141592654 * (i - 0.25) / (n + 0.5));
49  do {
50  p1 = 1.0;
51  p2 = 0.0;
52  for (j = 1; j <= n; j++) {
53  p3 = p2;
54  p2 = p1;
55  p1 = ((2.0 * j - 1.0) * z * p2 - (j - 1.0) * p3) / j;
56  }
57  pp = n * (z * p1 - p2) / (z * z - 1.0);
58  z1 = z;
59  z = z1 - p1 / pp;
60  } while (fabs(z - z1) > 3.0e-11);
61  x[i] = xm - xl * z;
62  x[n + 1 - i] = xm + xl * z;
63  w[i] = 2.0 * xl / ((1.0 - z * z) * pp * pp);
64  w[n + 1 - i] = w[i];
65  }
66  }

69  //containing the abscissas and weights of the Gauss-Chebyshev n-point quadrature formula.
70  void gaucheb(double x1, double x2, t_dArray1D &x, t_dArray1D &w, int n) {
71  double zr;
72  zr = 0.5 * (x2 - x1);
73  for (int i = 0; i <= n; i++) {
74  x[i] = 0.0;
75  w[i] = 0.0;
76  }
77  for (int i = 1; i <= n; i++) {
78  x[i] = zr * cos((i / (n + 1.0)) * pi);
79  w[i] = (pi / (n + 1.0)) * sin(i / (n + 1.0) * pi) * sin(i / (n + 1.0) * pi);
80  }

83  void setNodes(Integrator_ID _id) {
84  double aa, bb, zm, zr, dz;
85  switch (_id) {
86  case qgausleg_log_ID:
87  gauleg(0.0, 1.0, zz, w, NumPoints);
88  aa = log(LimDown);
89  bb = log(LimUp);
90  zm = (aa);
91  zr = (bb - aa);
92  for (int j = 1; j <= NumPoints; j++) {
93  dz = zr * zz[j];

95  w[j] = w[j] * exp(zm + dz) * zr;

97  break;
