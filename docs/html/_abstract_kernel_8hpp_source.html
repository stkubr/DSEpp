103  t_cmplx takeInnerProduct(t_cmplxTensor& A, t_cmplxTensor& B, int rank){
104  t_cmplx result=0.0;
105  if(rank==2){
106  for (int i = 0; i < 4; i++){
107  for (int j = 0; j < 4; j++){
108  result+=(A)(i,j)*(B)(j,i);} } }
109  if(rank==1){
110  for (int i = 0; i < 4; i++){
111  result+=(A)(i)*(B)(i);} }
112  if(rank==0){
113  result+=(A)(0)*(B)(0);}
